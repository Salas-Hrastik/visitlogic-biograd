import express from "express"; import fs from "fs"; import path from "path"; import OpenAI from "openai"; import fetch from "node-fetch"; const app = express(); const __dirname = path.resolve(); app.use(express.json()); app.use(express.static(__dirname)); const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY, }); function interpretWeather(code) { if ([0].includes(code)) return "sunčano"; if ([1,2,3].includes(code)) return "djelomično oblačno"; if ([45,48].includes(code)) return "magla"; if ([51,53,55,61,63,65].includes(code)) return "kiša"; if ([95,96,99].includes(code)) return "nevrijeme"; return "nepoznato"; } app.post("/api/chat", async (req, res) => { try { const { conversation } = req.body; const userMessage = conversation?.[conversation.length - 1]?.content?.toLowerCase() || ""; // 🌤 Vrijeme const weatherResponse = await fetch( "https://api.open-meteo.com/v1/forecast?latitude=43.9444&longitude=15.4444¤t_weather=true" ); const weatherData = await weatherResponse.json(); const temperature = weatherData.current_weather.temperature; const weatherDescription = interpretWeather( weatherData.current_weather.weathercode ); // 📂 Baza const filePath = path.join(__dirname, "data", "biograd_master.json"); const rawData = fs.readFileSync(filePath); const data = JSON.parse(rawData); let objekti = []; for (const kategorija in data.kategorije) { const items = data.kategorije[kategorija]; if (Array.isArray(items)) { items.forEach(item => { const score = ((item.ocjena || 0) * 2) + ((item.broj_ocjena || 0) / 100); objekti.push({ kategorija, score, ...item }); }); } } // 🔎 Detekcija kategorije let category = null; if ( userMessage.includes("restoran") || userMessage.includes("ručak") || userMessage.includes("večer") ) category = "restorani"; if (userMessage.includes("plaž")) category = "plaze"; if (userMessage.includes("ljekarn") || userMessage.includes("apotek")) category = "ljekarne"; if (userMessage.includes("hotel") || userMessage.includes("smještaj")) category = "smjestaj"; // 🎯 Strukturirani odgovor if (category) { const results = objekti .filter(o => o.kategorija === category && o.ocjena >= 4.2) .sort((a, b) => b.score - a.score) .slice(0, 3); return res.json({ type: "cards", title: `Preporuke (${weatherDescription}, ${temperature}°C)`, items: results.map(r => ({ naziv: r.naziv, adresa: r.adresa, ocjena: r.ocjena, opis: r.opis, lat: r.lat || null, lng: r.lng || null, web: r.web || null })) }); } // 🤖 AI fallback const completion = await openai.chat.completions.create({ model: "gpt-4o-mini", temperature: 0.3, messages: [ { role: "system", content: "Ti si profesionalni turistički informator za Biograd na Moru. Odgovaraj stručno i bez izmišljanja." }, { role: "user", content: userMessage } ] }); res.json({ reply: completion.choices[0].message.content }); } catch (error) { console.error(error); res.status(500).json({ reply: "Greška servera." }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
